cmake_minimum_required(VERSION 3.15.5)

project(mve_texrecon VERSION 0.1.0 DESCRIPTION "mve + texrecon (MultiView texturing subset for ASP)")

# Trimmed from NeoGeographyToolkit/MultiView top CMakeLists: build only the
# texturing stack (mve, rayint, texrecon) needed by the Ames Stereo Pipeline.
# mapmap is header-only for texrecon (its own CMakeLists builds test/demo
# executables we do not need), so it is NOT add_subdirectory'd here - texrecon
# includes its headers directly via ${CMAKE_SOURCE_DIR}/mapmap. voxblox,
# TheiaSfM and cgal_tools are separate feedstocks.

if("${MULTIVIEW_DEPS_DIR}" STREQUAL "")
  message(FATAL_ERROR "Set MULTIVIEW_DEPS_DIR as the path to a conda env having the dependencies.")
endif()

set(CMAKE_BUILD_TYPE "Release")

# Where to search for dependencies (conda env)
set(CMAKE_PREFIX_PATH "${MULTIVIEW_DEPS_DIR};${CMAKE_PREFIX_PATH}")
include_directories("${MULTIVIEW_DEPS_DIR}/include")
set(CMAKE_INSTALL_LIBDIR "lib") # install in lib, not lib64

# mve: builds mveCore/mveUtil/mveDmrecon shared libs + installs mve headers.
add_subdirectory(mve)

# rayint: header-only ray tracing; installs rayint headers (also builds a small
# raycast app that links mveCore, which is available from the mve subdir above).
add_subdirectory(rayint)

# texrecon (mvs-texturing): builds the texrecon binary + texture_reconstruction
# lib + installs texrecon headers. It finds its own TBB/OpenMP/PNG/JPEG/TIFF/
# Eigen3 and references ${CMAKE_SOURCE_DIR}/{mve,rayint,mapmap} for headers, so
# it must be built after mve/rayint and with mapmap present as a sibling.
add_subdirectory(texrecon)
